#include <stdio.h>
#include <string.h>

int main() {
    char str[100];
    printf("请输入字符串:");
    gets(str);

    int len = strlen(str);
    char same = str[0];
    int count = 1;

    printf("结果为:");
    for (int i = 1; i < len; i++) {
        if (str[i] == same) {
            count++; 
        } else {
            printf("%c%d", same, count);
            same = str[i];
            count = 1; 
        }
    }
    printf("%c%d\n", same, count);

    return 0;
}
